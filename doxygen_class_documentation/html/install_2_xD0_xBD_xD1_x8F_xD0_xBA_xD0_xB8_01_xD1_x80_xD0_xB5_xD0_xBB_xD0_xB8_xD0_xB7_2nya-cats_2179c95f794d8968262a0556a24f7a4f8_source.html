340 void LGen2DiagramEditor::setCenter(const QPointF& centerPoint) {
341  // Получаем границы видимой области в координатах сцены
342  QRectF visibleArea = mapToScene(rect()).boundingRect();
343  // Получаем границы сцены
344  QRectF sceneBounds = sceneRect();
345  // Подсчитываем максимальные границы, в которых должен быть центр
346 // double boundX = visibleArea.width() / 2.0;
347 // double boundY = visibleArea.height() / 2.0;
348 // double boundWidth = sceneBounds.width() - visibleArea.width();
349 // double boundHeight = sceneBounds.height() - visibleArea.height();
350 
351  double boundX = sceneBounds.x()+ visibleArea.width() / 2.0;
352  double boundY = sceneBounds.y()+ visibleArea.height() / 2.0;
353  double boundWidth = sceneBounds.width() - visibleArea.width();
354  double boundHeight = sceneBounds.height() - visibleArea.height();
355 
356  QRectF bounds(boundX, boundY, boundWidth, boundHeight);
357  // Допускаем, что мы попали в границы, тогда все ок
358  center = centerPoint;
359  // А если не попали, то корректируем центр
360  if(!bounds.contains(centerPoint))
361  if(visibleArea.contains(sceneBounds))
362  {
363  // Случай, если видна вся сцена
364  center = sceneBounds.center();
365  }
366  else
367  {
368  // Сдвигаем центр к нужным границам
369  if(centerPoint.x() > bounds.x() + bounds.width())
370  center.setX(bounds.x() + bounds.width());
371  else if(centerPoint.x() < bounds.x())
372  center.setX(bounds.x());
373  if(centerPoint.y() > bounds.y() + bounds.height())
374  center.setY(bounds.y() + bounds.height());
375  else if(centerPoint.y() < bounds.y())
376  center.setY(bounds.y());
377  }
378  // Обновляем полосы прокрутки
379  centerOn(center);
380 }
