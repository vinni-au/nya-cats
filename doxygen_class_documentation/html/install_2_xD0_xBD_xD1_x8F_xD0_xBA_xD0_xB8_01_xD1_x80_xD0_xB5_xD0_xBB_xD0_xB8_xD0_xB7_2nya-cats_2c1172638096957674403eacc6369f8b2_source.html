235  QString domainName = DataModels::instance()->domainModel->data(firstDomainIndex,Qt::DisplayRole).toString();
236

240  QModelIndex askingIndex = DataModels::instance()->varModel->index(0,0,varIndex);

250  DataModels::instance()->varModel->setData(derivableIndex,true,Qt::EditRole);

252 
253  //обновить домен
254  onCurrentVarChanged(varIndex.row());
255  }
