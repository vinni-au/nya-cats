560 NKBManager::readFromXml(QFile &file)
561 {
562  doc.clear();
563  if(!doc.setContent(&file))
564  {
565  qDebug()<<"NKBManager::readFromXml: Не удалось считать файл: "<<file.fileName();
566  return;
567  }
568 
569  QDomElement kbEl = doc.documentElement();
570  //домены
571  QDomElement domainsEl = kbEl.firstChildElement("domains");
572  m_domainModel.fromXml(domainsEl);
573  //фреймы
574  QDomElement framesEl = kbEl.firstChildElement("frames");
575  QDomElement frameEl = framesEl.firstChild().toElement();
576  while(!frameEl.isNull())
577  {
578  NFrame *frame = new NFrame(0);
579  frame->fromXml(frameEl);
580  m_frames.append(frame);
581  frameEl = frameEl.nextSibling().toElement();
582  }
583  getFrameNetModel()->setFrames(&m_frames);
584  //процедуры
585  QDomElement procsEl = kbEl.firstChildElement("procs");
586  QDomElement procEl = procsEl.firstChild().toElement();
587  while(!procEl.isNull())
588  {
589  NProc *proc = new NProc();
590  proc->fromXml(procEl);
591  m_procs.append(proc);
592  procEl = procEl.nextSibling().toElement();
593  }
594  //продукции
595  QDomElement productionsEl = kbEl.firstChildElement("productions");
596  QDomElement prodictionEl = productionsEl.firstChild().toElement();
597  while(!prodictionEl.isNull())
598  {
599  NProduction *production = new NProduction();
600  production->fromXml(prodictionEl);
601  m_productions.append(production);
602  prodictionEl = prodictionEl.nextSibling().toElement();
603  }
604  //диаграммы
605  m_diagramElement=kbEl.firstChildElement("diagram");
606 
607  //глобальные процедуры
608  QDomElement globalProcsEl = kbEl.firstChildElement("global_procs");
609  if(!globalProcsEl.isNull())
610  {
611  setGlobalProcsContext( globalProcsEl.text() );
612  }
613  file.close();
614  setDirty(false);
615  emit sigDataLoaded();
616 }
