178 StartConsWnd::ruleAccepted(Rule *rule)
179 {
180 //а у меня ленивые вычисления условий?
181  //разбираем посылку
182 
183  Expr* e;
184  for(int i=0;i<rule->predicate.count();i++)
185  {
186  if(!continueWork)
187  return false;
188  e = rule->predicate.at(i);
189  qDebug()<<"Current rule predicate: "<<rule->name<<" "<<e->toString();
190  if(ConsVars.contains(e->var))//если такая переменная уже означена
191  {
192  qDebug()<<"Var exists: "<<e->var<<" value: "<<ConsVars[e->var];
193  if(e->sign == "=")
194  {
195  if(! (e->val == ConsVars[e->var]))//не то значение

201  }
